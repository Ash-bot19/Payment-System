    ```python
    """Nightly reconciliation DAG — Phase 07.

    Reconciles internal ledger_entries against Stripe PaymentIntents for the
    midnight-to-midnight UTC window derived from Airflow's {{ ds }} variable.

    DAG ID: nightly_reconciliation
    Schedule: @daily (midnight UTC)
    Start date: 2026-01-01 UTC (deterministic, not datetime.now())

    Task dependency graph:
        detect_duplicates ──\
                             ──> compare_and_publish
        fetch_stripe_window ─/

    detect_duplicates and fetch_stripe_window run in parallel (Wave 1).
    compare_and_publish depends on fetch_stripe_window output (Wave 2).

    Idempotency: same window = same source data (per D-08). Duplicate Kafka
    messages are accepted; Phase 8 dbt mart uses COUNT DISTINCT / ROW_NUMBER.
    """

    import os
    from datetime import date, datetime, timezone
    from typing import Any

    import pendulum
    import structlog
    import stripe
    from airflow.decorators import dag, task
    from airflow.utils.dates import days_ago
    from sqlalchemy import create_engine, text

    from kafka.producers.reconciliation_producer import ReconciliationProducer
    from models.reconciliation import DiscrepancyType, ReconciliationMessage

    logger = structlog.get_logger(__name__)

    stripe.max_network_retries = 3  # Built-in SDK exponential backoff (per D-17)

    BOOTSTRAP_SERVERS = os.getenv("KAFKA_BOOTSTRAP_SERVERS", "localhost:9092")
    DATABASE_URL_SYNC = os.getenv(
        "DATABASE_URL_SYNC",
        "postgresql://payment:payment@localhost:5432/payment_db",
    )


    @dag(
        dag_id="nightly_reconciliation",
        schedule="@daily",
        start_date=pendulum.datetime(2026, 1, 1, tz="UTC"),
        catchup=False,
        tags=["reconciliation", "phase-07"],
    )
    def nightly_reconciliation_dag():
        @task()
        def detect_duplicates(data_interval_start: Any = None, data_interval_end: Any = None, ds: str = None) -> int:
            """Type 4: detect transaction_ids with > 2 ledger rows. Pure SQL, no Stripe call."""
            return detect_duplicates_fn(
                window_start=data_interval_start,
                window_end=data_interval_end,
                run_date=date.fromisoformat(ds),
                db_url=DATABASE_URL_SYNC,
                bootstrap_servers=BOOTSTRAP_SERVERS,
            )

        @task()
        def fetch_stripe_window(data_interval_start: Any = None, data_interval_end: Any = None) -> dict:
            """Fetch all succeeded PaymentIntents for the window from Stripe API."""
            stripe.api_key = os.getenv("STRIPE_SECRET_KEY", "")
            return fetch_stripe_window_fn(
                window_start=data_interval_start,
                window_end=data_interval_end,
            )

        @task()
        def compare_and_publish(stripe_intents: dict, data_interval_start: Any = None, data_interval_end: Any = None, ds: str = None) -> int:
            """Type 1 + Type 3: compare ledger vs Stripe, publish discrepancies."""
            return compare_and_publish_fn(
                stripe_intents=stripe_intents,
                window_start=data_interval_start,
                window_end=data_interval_end,
                run_date=date.fromisoformat(ds),
                db_url=DATABASE_URL_SYNC,
                bootstrap_servers=BOOTSTRAP_SERVERS,
            )

        stripe_data = fetch_stripe_window()
        detect_duplicates()
        compare_and_publish(stripe_intents=stripe_data)


    nightly_reconciliation_dag()
    ```

    detect_duplicates_fn(window_start, window_end, run_date, db_url, bootstrap_servers) -> int:
    - Creates SQLAlchemy engine, executes:
      SELECT transaction_id, merchant_id, MIN(currency) as currency
      FROM ledger_entries
      WHERE created_at >= :window_start AND created_at < :window_end
      GROUP BY transaction_id, merchant_id
      HAVING COUNT(*) > 2
    - For each row: publishes ReconciliationMessage(transaction_id=row.transaction_id, discrepancy_type=DiscrepancyType.DUPLICATE_LEDGER, currency=row.currency, merchant_id=row.merchant_id, run_date=run_date, stripe_payment_intent_id=None, internal_amount_cents=None, stripe_amount_cents=None, diff_cents=None)
    - Uses ReconciliationProducer(bootstrap_servers); calls producer.close() in finally block
    - Returns count of published messages
    - Logs: structlog.bind(run_date=run_date, window_start=window_start, window_end=window_end)
