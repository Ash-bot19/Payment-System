    TASK 1 — detect_duplicates (inside DAG function):
    ```python
    @task
    def detect_duplicates(ds: str = None) -> int:
        window_start, window_end = parse_window(ds)
        run_date = datetime.strptime(ds, "%Y-%m-%d").date()
        log = logger.bind(task="detect_duplicates", run_date=str(run_date),
                          window_start=window_start.isoformat(), window_end=window_end.isoformat())

        engine = create_engine(os.environ["DATABASE_URL_SYNC"])
        producer = ReconciliationProducer(bootstrap_servers=os.environ["KAFKA_BOOTSTRAP_SERVERS"])

        sql = text("""
            SELECT transaction_id, COUNT(*) as row_count, MAX(merchant_id) as merchant_id
            FROM ledger_entries
            WHERE created_at >= :window_start AND created_at < :window_end
            GROUP BY transaction_id
            HAVING COUNT(*) > 2
        """)

        count = 0
        try:
            with engine.connect() as conn:
                rows = conn.execute(sql, {"window_start": window_start, "window_end": window_end}).fetchall()
            for row in rows:
                msg = ReconciliationMessage(
                    transaction_id=row.transaction_id,
                    discrepancy_type=DiscrepancyType.DUPLICATE_LEDGER,
                    stripe_payment_intent_id=None,  # per D-12: redundant with transaction_id
                    internal_amount_cents=None,
                    stripe_amount_cents=None,
                    diff_cents=None,
                    currency="USD",
                    merchant_id=row.merchant_id,
                    stripe_created_at=None,
                    run_date=run_date,
                )
                producer.publish(row.transaction_id, msg.model_dump(mode="json"))
                count += 1
                log.info("duplicate_ledger_published", transaction_id=row.transaction_id, row_count=row.row_count)
        finally:
            producer.close()
            engine.dispose()
        log.info("detect_duplicates_complete", published=count)
        return count
    ```

    TASK 2 — fetch_stripe_window (inside DAG function):
    ```python
    @task
    def fetch_stripe_window(ds: str = None) -> dict[str, Any]:
        window_start, window_end = parse_window(ds)
        log = logger.bind(task="fetch_stripe_window", run_date=ds,
                          window_start=window_start.isoformat(), window_end=window_end.isoformat())

        stripe_window: dict[str, Any] = {}
        page_iter = stripe.PaymentIntent.list(
            created={"gte": int(window_start.timestamp()), "lt": int(window_end.timestamp())},
            limit=100,
        ).auto_paging_iter()

        for pi in page_iter:
            if pi["status"] != "succeeded":  # per D-15: filter to succeeded only
                continue
            stripe_window[pi["id"]] = {
                "stripe_payment_intent_id": pi["id"],
                "stripe_amount_cents": pi["amount_received"],
                "currency": pi.get("currency", "usd").upper(),
                "merchant_id": pi.get("metadata", {}).get("merchant_id", "unknown"),
                "stripe_created_at": datetime.fromtimestamp(pi["created"], tz=timezone.utc).isoformat(),
            }
        log.info("fetch_stripe_window_complete", fetched=len(stripe_window))
        return stripe_window  # XCom return value consumed by compare_and_publish
    ```

    ```python
    def compare_and_publish_fn(
        stripe_window: dict[str, Any],
        ds: str,
        engine,           # sqlalchemy Engine (injected for testability)
        producer: ReconciliationProducer,
    ) -> int:
        window_start, window_end = parse_window(ds)
        run_date = datetime.strptime(ds, "%Y-%m-%d").date()
        log = logger.bind(task="compare_and_publish", run_date=str(run_date))

        # Fetch ledger window: one DEBIT row per transaction_id (amount is the same for DEBIT/CREDIT)
        sql = text("""
            SELECT DISTINCT ON (transaction_id)
                transaction_id,
                amount_cents,
                merchant_id,
                currency
            FROM ledger_entries
            WHERE created_at >= :window_start AND created_at < :window_end
              AND entry_type = 'DEBIT'
            ORDER BY transaction_id, created_at ASC
        """)

        count = 0
        with engine.connect() as conn:
            ledger_rows = conn.execute(sql, {"window_start": window_start, "window_end": window_end}).fetchall()

        ledger_by_txn: dict[str, Any] = {row.transaction_id: row for row in ledger_rows}

        # Type 1: MISSING_INTERNALLY — in Stripe, not in ledger
        for txn_id, stripe_data in stripe_window.items():
            if txn_id not in ledger_by_txn:
                msg = ReconciliationMessage(
                    transaction_id=txn_id,
                    discrepancy_type=DiscrepancyType.MISSING_INTERNALLY,
                    stripe_payment_intent_id=stripe_data["stripe_payment_intent_id"],
                    internal_amount_cents=None,
                    stripe_amount_cents=stripe_data["stripe_amount_cents"],
                    diff_cents=None,  # no internal row to diff against per D-11
                    currency=stripe_data.get("currency", "USD"),
                    merchant_id=stripe_data.get("merchant_id", "unknown"),
                    stripe_created_at=datetime.fromisoformat(stripe_data["stripe_created_at"]) if stripe_data.get("stripe_created_at") else None,
                    run_date=run_date,
                )
                producer.publish(txn_id, msg.model_dump(mode="json"))
                count += 1
                log.info("missing_internally_published", transaction_id=txn_id)

        # Type 3: AMOUNT_MISMATCH — in both, amounts differ
        for txn_id, ledger_row in ledger_by_txn.items():
            if txn_id in stripe_window:
                stripe_amount = stripe_window[txn_id]["stripe_amount_cents"]
                internal_amount = ledger_row.amount_cents
                if stripe_amount != internal_amount:
                    msg = ReconciliationMessage(
                        transaction_id=txn_id,
                        discrepancy_type=DiscrepancyType.AMOUNT_MISMATCH,
                        stripe_payment_intent_id=stripe_window[txn_id]["stripe_payment_intent_id"],
                        internal_amount_cents=internal_amount,
                        stripe_amount_cents=None,  # per D-11: null for Type 3 stripe_amount_cents
                        diff_cents=stripe_amount - internal_amount,
                        currency=ledger_row.currency,
                        merchant_id=ledger_row.merchant_id,
                        stripe_created_at=None,
                        run_date=run_date,
                    )
                    producer.publish(txn_id, msg.model_dump(mode="json"))
                    count += 1
                    log.info("amount_mismatch_published", transaction_id=txn_id,
                             internal=internal_amount, stripe=stripe_amount, diff=stripe_amount - internal_amount)

        log.info("compare_and_publish_complete", published=count)
        return count
    ```
