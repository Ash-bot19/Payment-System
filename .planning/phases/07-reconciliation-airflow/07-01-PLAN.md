class AlertProducer:
    def __init__(self, bootstrap_servers: str) -> None:
        self._producer = Producer({"bootstrap.servers": bootstrap_servers})

    def publish(self, stripe_event_id: str, scored_event: dict[str, Any]) -> None:
        backoff_seconds = [1, 2, 4]
        last_exc: KafkaException | None = None
        for attempt, backoff in enumerate(backoff_seconds, start=1):
            try:
                self._producer.produce(
                    topic=TOPIC,
                    key=stripe_event_id.encode("utf-8"),
                    value=json.dumps(scored_event, default=str).encode("utf-8"),
                    on_delivery=self._delivery_report,
                )
                self._producer.flush(timeout=5)
                return
            except KafkaException as exc:
                last_exc = exc
                if attempt < len(backoff_seconds):
                    time.sleep(backoff)
        raise last_exc

    @staticmethod
    def _delivery_report(err, msg) -> None: ...

    def close(self) -> None:
        self._producer.flush()
```

    """Pydantic model for payment.reconciliation.queue messages — Phase 07.

    ReconciliationMessage: One message per discrepancy, published by the
    nightly_reconciliation Airflow DAG. Schema locked per D-11 in 07-CONTEXT.md.

    DiscrepancyType enum values:
      MISSING_INTERNALLY — Stripe has succeeded PaymentIntent; no ledger row
      AMOUNT_MISMATCH    — Both sides have record; amount_cents differs
      DUPLICATE_LEDGER   — More than 2 ledger rows for one transaction_id
    """

    import enum
    from datetime import date, datetime
    from typing import Optional

    from pydantic import BaseModel, ConfigDict


    class DiscrepancyType(str, enum.Enum):
        MISSING_INTERNALLY = "MISSING_INTERNALLY"
        AMOUNT_MISMATCH = "AMOUNT_MISMATCH"
        DUPLICATE_LEDGER = "DUPLICATE_LEDGER"


    class ReconciliationMessage(BaseModel):
        model_config = ConfigDict(use_enum_values=True)

        transaction_id: str
        discrepancy_type: DiscrepancyType
        stripe_payment_intent_id: Optional[str] = None   # null for Type 4 (D-12)
        internal_amount_cents: Optional[int] = None       # null for Type 1
        stripe_amount_cents: Optional[int] = None         # null for Type 3/4
        diff_cents: Optional[int] = None                  # null for Type 1/4; stripe-internal for Type 3
        currency: str                                      # always present; "USD" in MVP
        merchant_id: str
        stripe_created_at: Optional[datetime] = None      # Stripe PaymentIntent created_at; null for Type 3/4
        run_date: date                                     # Airflow {{ ds }}; enables BigQuery partitioning
    ```
