    ```python
    @pytest.mark.e2e
    def test_reconciliation_kafka_roundtrip():
        """Publish a ReconciliationMessage to Kafka and consume it back."""
        bootstrap = os.environ.get("KAFKA_BOOTSTRAP_SERVERS", "localhost:9092")
        try:
            from confluent_kafka.admin import AdminClient
            AdminClient({"bootstrap.servers": bootstrap}).list_topics(timeout=3)
        except Exception:
            pytest.skip("Kafka not available — skipping E2E test")

        from kafka.producers.reconciliation_producer import ReconciliationProducer
        txn_id = f"pi_e2e_{uuid.uuid4().hex[:8]}"
        msg = ReconciliationMessage(
            transaction_id=txn_id,
            discrepancy_type=DiscrepancyType.DUPLICATE_LEDGER,
            currency="USD",
            merchant_id="m_e2e",
            run_date=date(2024, 1, 15),
        )
        producer = ReconciliationProducer(bootstrap_servers=bootstrap)
        producer.publish(txn_id, msg.model_dump(mode="json"))
        producer.close()

        consumer = Consumer({
            "bootstrap.servers": bootstrap,
            "group.id": f"e2e-recon-{uuid.uuid4().hex[:8]}",
            "auto.offset.reset": "earliest",
            "enable.auto.commit": True,
        })
        consumer.subscribe(["payment.reconciliation.queue"])
        consumed = None
        for _ in range(30):
            record = consumer.poll(timeout=1.0)
            if record and not record.error():
                payload = json.loads(record.value())
                if payload.get("transaction_id") == txn_id:
                    consumed = payload
                    break
        consumer.close()

        assert consumed is not None, f"Message for {txn_id} not found on payment.reconciliation.queue"
        validated = ReconciliationMessage(**consumed)
        assert validated.discrepancy_type == DiscrepancyType.DUPLICATE_LEDGER
        assert validated.stripe_payment_intent_id is None  # per D-12
        assert validated.run_date == date(2024, 1, 15)
    ```
