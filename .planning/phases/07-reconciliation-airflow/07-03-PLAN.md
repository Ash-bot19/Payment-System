    ```
    # Switch to root to install system packages, then back to airflow user
    USER root
    RUN apt-get update && apt-get install -y --no-install-recommends \
        libpq-dev gcc && rm -rf /var/lib/apt/lists/*
    USER airflow

    # Copy project source for DAG imports
    COPY --chown=airflow:root payment-backend/ /app/

    WORKDIR /app

    # Install project dependencies (confluent-kafka, structlog, stripe, sqlalchemy, pydantic, etc.)
    RUN pip install --no-cache-dir \
        apache-airflow-providers-common-sql \
        confluent-kafka==2.3.0 \
        structlog \
        stripe \
        sqlalchemy \
        psycopg2-binary \
        pydantic

    ENV PYTHONPATH=/app
    ENV AIRFLOW__CORE__DAGS_FOLDER=/app/airflow/dags
    ENV AIRFLOW__CORE__LOAD_EXAMPLES=false
    ENV AIRFLOW__CORE__EXECUTOR=LocalExecutor
    ```

    Add airflow service to docker-compose.yml (append before the final `volumes:` block):
    ```yaml
      # --- Airflow (nightly reconciliation DAG) ---
      airflow:
        build:
          context: ..
          dockerfile: infra/Dockerfile.airflow
        container_name: airflow
        ports:
          - "8080:8080"
        environment:
          - AIRFLOW__CORE__SQL_ALCHEMY_CONN=postgresql+psycopg2://payment:payment@postgres:5432/airflow_db
          - AIRFLOW__CORE__FERNET_KEY=${AIRFLOW_FERNET_KEY:-$(python3 -c "from cryptography.fernet import Fernet; print(Fernet.generate_key().decode())")}
          - AIRFLOW__CORE__DAGS_FOLDER=/app/airflow/dags
          - AIRFLOW__CORE__LOAD_EXAMPLES=false
          - AIRFLOW__CORE__EXECUTOR=LocalExecutor
          - KAFKA_BOOTSTRAP_SERVERS=kafka:29092
          - DATABASE_URL_SYNC=postgresql://payment:payment@postgres:5432/payment_db
          - STRIPE_SECRET_KEY=${STRIPE_SECRET_KEY:-sk_test_placeholder}
        depends_on:
          postgres:
            condition: service_healthy
          kafka:
            condition: service_healthy
        command: >
          bash -c "
            airflow db migrate &&
            airflow users create --username admin --password admin --firstname Admin --lastname User --role Admin --email admin@example.com || true &&
            airflow scheduler &
            airflow webserver -p 8080
          "
        healthcheck:
          test: ["CMD", "curl", "-f", "http://localhost:8080/health"]
          interval: 30s
          timeout: 10s
          retries: 5
          start_period: 60s
        restart: unless-stopped
    ```

    IMPORTANT: Airflow needs the airflow_db PostgreSQL database to exist. Add an init SQL script or note that the user must CREATE DATABASE airflow_db before first run. The cleanest solution is to add an `airflow-db-init` one-shot service:
    ```yaml
      airflow-db-init:
        image: postgres:16-alpine
        container_name: airflow-db-init
        environment:
          PGPASSWORD: payment
        command: >
          bash -c "psql -h postgres -U payment -d payment_db -c 'CREATE DATABASE airflow_db;' || echo 'airflow_db already exists'"
        depends_on:
          postgres:
            condition: service_healthy
        restart: "no"
    ```
    Make airflow service depend on airflow-db-init completing (add `airflow-db-init` to its depends_on without a condition — one-shot containers exit 0).
