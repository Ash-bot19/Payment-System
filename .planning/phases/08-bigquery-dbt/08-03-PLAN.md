def test_assert_ledger_balanced_fails_on_imbalanced_data():
    """assert_ledger_balanced singular test exits 1 when imbalanced rows exist.

    Failure path: Insert DEBIT=10000 and CREDIT=9999 into ledger_entries for
    the same transaction_id. This bypasses the balance trigger by inserting
    each row in its own transaction (not in a paired transaction).
    dbt rebuilds fact_ledger_balanced and the singular test should fail.

    NOTE: The balance trigger (DEFERRABLE INITIALLY DEFERRED) fires at COMMIT
    time within a single transaction. Inserting each row in its own separate
    connection/transaction bypasses the trigger because each individual insert
    is trivially balanced at commit time (the trigger computes balance per row,
    and a single DEBIT row alone is not checked against CREDIT).
    Actually, the trigger runs per row and checks current sum — so two separate
    transactions for DEBIT and CREDIT will each individually fail the check.
    Use a direct psycopg2 connection with autocommit=True per row to bypass.
    """
    tx_id = "e2e_imbalanced_tx_001"
    conn = psycopg2.connect(DATABASE_URL)
    conn.autocommit = True
    cur = conn.cursor()

    # Insert DEBIT in its own autocommit transaction (trigger sees sum=10000-0=10000 != 0 -> FAIL)
    # BUT: the trigger checks SUM(CREDIT) - SUM(DEBIT) = 0. With only DEBIT inserted,
    # CREDIT sum = 0, DEBIT sum = 10000, balance = -10000 != 0 — trigger FIRES and raises.
    # To bypass: disable the trigger temporarily (requires superuser) or use
    # SET session_replication_role = replica (also superuser).
    # Simplest approach for test: insert via psycopg2 with SET CONSTRAINTS DEFERRED
    # in a single transaction but with different amounts.
    conn.autocommit = False
    try:
        cur.execute("SET CONSTRAINTS check_ledger_balance DEFERRED")
        cur.execute(
            """INSERT INTO ledger_entries
               (transaction_id, amount_cents, entry_type, merchant_id,
                currency, source_event_id)
               VALUES (%s, 10000, 'DEBIT', 'merch_e2e_001', 'USD', %s)""",
            (tx_id, f"evt_{tx_id}_debit"),
        )
        cur.execute(
            """INSERT INTO ledger_entries
               (transaction_id, amount_cents, entry_type, merchant_id,
                currency, source_event_id)
               VALUES (%s, 9999, 'CREDIT', 'merch_e2e_001', 'USD', %s)""",
            (tx_id, f"evt_{tx_id}_credit"),
        )
        # SET CONSTRAINTS IMMEDIATE forces the trigger to fire now (before COMMIT)
        # but since the balance is -1 != 0, this will raise — which is what we DON'T want.
        # Instead: commit WITHOUT setting immediate — DEFERRABLE INITIALLY DEFERRED
        # means it fires at COMMIT. The trigger WILL raise at commit.
        # So we cannot actually commit imbalanced data via normal INSERT.
        # Correct approach: use a different transaction_id and insert via
        # SET session_replication_role = replica to disable triggers.
        # Per project permissions: use a direct pg_catalog approach.
        # Simplest safe approach: use a seed-based fact_ledger_balanced override.
