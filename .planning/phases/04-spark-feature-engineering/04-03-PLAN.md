    Create payment-backend/tests/integration/spark/test_spark_kafka_redis.py.

    This test simulates the end-to-end pipeline WITHOUT a running Kafka broker by creating a static DataFrame that mimics what the Kafka readStream would produce, then running the enrichment and foreachBatch logic directly.

    APPROACH: Instead of requiring a live Kafka broker (which makes CI fragile), this test:
    a. Creates a SparkSession (local[2])
    b. Builds a DataFrame matching the enriched schema (event_id, merchant_id, stripe_customer_id, amount_cents, received_at, hour_of_day, weekend_flag, amount_cents_log)
    c. Calls write_features_to_redis(batch_df, epoch_id=0) directly — the same foreachBatch function the streaming job uses
    d. Asserts the Redis hash exists with correct keys and values

    This requires a LIVE Redis instance (redis://localhost:6379/0). Mark the test with @pytest.mark.integration. If Redis is not available, the test should skip gracefully with pytest.importorskip or a connection check.

    STRUCTURE:
    ```python
    import os
    import math
    import json
    import pytest
    import redis
    import structlog
    from datetime import datetime
    from pyspark.sql import SparkSession, functions as F
    from pyspark.sql.types import (
        StructType, StructField, StringType, LongType, TimestampType, IntegerType, DoubleType
    )

    # Feature keys per CLAUDE.md ML Risk Score Contract
    EXPECTED_FEATURE_KEYS = [
        "tx_velocity_1m", "tx_velocity_5m", "amount_zscore",
        "merchant_risk_score", "device_switch_flag",
        "hour_of_day", "weekend_flag", "amount_cents_log",
    ]

    @pytest.fixture(scope="module")
    def redis_client():
        """Connect to local Redis. Skip all tests in this module if Redis unavailable."""
        url = os.getenv("REDIS_URL", "redis://localhost:6379/0")
        r = redis.Redis.from_url(url, decode_responses=True)
        try:
            r.ping()
        except redis.ConnectionError:
            pytest.skip("Redis not available at " + url)
        yield r

    @pytest.fixture(autouse=True)
    def clean_redis_keys(redis_client):
        """Clean up test keys after each test."""
        yield
        # Delete test keys
        for key in redis_client.scan_iter("feat:evt_test_*"):
            redis_client.delete(key)
        for key in redis_client.scan_iter("merchant:stats:merch_test_*"):
            redis_client.delete(key)
        for key in redis_client.scan_iter("device_seen:cus_test_*"):
            redis_client.delete(key)
    ```
