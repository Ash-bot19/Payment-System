Create `payment-backend/scripts/send_webhook_events.py`.

This script:
1. Generates N synthetic `payment_intent.succeeded` Stripe-format events
2. Signs each event with STRIPE_WEBHOOK_SECRET using `stripe.WebhookSignature`
3. POSTs each to the deployed webhook-service URL
4. Logs success/failure with structlog (never print())
5. Prints a summary at the end

**Exact implementation:**

```python
"""Synthetic webhook event sender — Phase 11 Loom demo.

Sends N signed Stripe-format payment_intent.succeeded events to the
deployed webhook-service Cloud Run URL.

Usage:
    python scripts/send_webhook_events.py \
        --url https://webhook-service-xxx-el.a.run.app \
        --secret whsec_xxx \
        --count 50

The script uses stripe.WebhookSignature to sign each event, ensuring the
webhook-service HMAC verification passes identically to live Stripe events.
"""

import argparse
import json
import random
import time
import uuid
from datetime import datetime, timezone

import requests
import stripe
import structlog

logger = structlog.get_logger(__name__)

MERCHANTS = ["merchant_a", "merchant_b", "merchant_c"]
EVENT_TYPES = ["payment_intent.succeeded"] * 8 + [
    "payment_intent.payment_failed",
    "payment_intent.canceled",
]


def build_event(event_type: str = "payment_intent.succeeded") -> dict:
    """Build a Stripe-format webhook event payload."""
    pi_id = f"pi_{uuid.uuid4().hex[:24]}"
    amount = random.randint(500, 50000)  # $5 to $500 in cents
    merchant_id = random.choice(MERCHANTS)

    return {
        "id": f"evt_{uuid.uuid4().hex[:24]}",
        "object": "event",
        "api_version": "2024-04-10",
        "created": int(datetime.now(tz=timezone.utc).timestamp()),
        "type": event_type,
        "data": {
            "object": {
                "id": pi_id,
                "object": "payment_intent",
                "amount": amount,
                "currency": "usd",
                "status": "succeeded" if event_type == "payment_intent.succeeded" else "canceled",
                "customer": f"cus_{uuid.uuid4().hex[:14]}",
                "metadata": {
                    "merchant_id": merchant_id,
                    "device_id": f"device_{random.randint(1, 100)}",
                },
                "created": int(datetime.now(tz=timezone.utc).timestamp()),
            }
        },
        "livemode": False,
    }


def sign_payload(payload_bytes: bytes, secret: str) -> str:
    """Generate Stripe-Signature header value using stripe SDK."""
    timestamp = int(time.time())
    signed_payload = f"{timestamp}.{payload_bytes.decode('utf-8')}"
    sig = stripe.WebhookSignature._compute_signature(signed_payload, secret)
    return f"t={timestamp},v1={sig}"


def send_events(url: str, secret: str, count: int, delay_ms: int = 100) -> None:
    """Send count webhook events to url, signed with secret."""
    endpoint = url.rstrip("/") + "/webhook"
    success = 0
    duplicate = 0
    failed = 0

    for i in range(count):
        event_type = random.choice(EVENT_TYPES)
        event = build_event(event_type)
        payload = json.dumps(event).encode("utf-8")
        sig_header = sign_payload(payload, secret)

        try:
            resp = requests.post(
                endpoint,
                data=payload,
                headers={
                    "Content-Type": "application/json",
                    "Stripe-Signature": sig_header,
                },
                timeout=10,
            )
            if resp.status_code == 200:
                success += 1
                logger.info(
                    "event_sent",
                    index=i + 1,
                    event_id=event["id"],
                    event_type=event_type,
                    status=resp.status_code,
                )
            elif resp.status_code == 200 and "duplicate" in resp.text:
                duplicate += 1
            else:
                failed += 1
                logger.warning(
                    "event_rejected",
                    index=i + 1,
                    event_id=event["id"],
                    status=resp.status_code,
                    body=resp.text[:200],
                )
        except Exception as exc:
            failed += 1
            logger.error("event_error", index=i + 1, error=str(exc))

        if delay_ms > 0:
            time.sleep(delay_ms / 1000)

    logger.info(
        "send_complete",
        total=count,
        success=success,
        duplicate=duplicate,
        failed=failed,
    )


def main() -> None:
    """CLI entrypoint."""
    parser = argparse.ArgumentParser(description="Send synthetic Stripe webhook events")
    parser.add_argument("--url", required=True, help="Webhook service base URL")
    parser.add_argument("--secret", required=True, help="STRIPE_WEBHOOK_SECRET value")
    parser.add_argument("--count", type=int, default=50, help="Number of events to send (default: 50)")
    parser.add_argument("--delay-ms", type=int, default=100, help="Delay between events in ms (default: 100)")
    args = parser.parse_args()

    structlog.configure(
        processors=[
            structlog.stdlib.add_log_level,
            structlog.dev.ConsoleRenderer(),
        ]
    )

    logger.info("starting", url=args.url, count=args.count, delay_ms=args.delay_ms)
    send_events(url=args.url, secret=args.secret, count=args.count, delay_ms=args.delay_ms)


if __name__ == "__main__":
    main()
```
