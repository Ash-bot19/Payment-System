**Step A — Create the Compute Engine VM:**
```bash
gcloud compute instances create kafka-vm \
  --project=project-2f9d2775-493e-4e59-9b8 \
  --zone=asia-south1-a \
  --machine-type=e2-standard-2 \
  --image-family=ubuntu-2204-lts \
  --image-project=ubuntu-os-cloud \
  --boot-disk-size=30GB \
  --network=default \
  --scopes=cloud-platform \
  --metadata=startup-script='#!/bin/bash
    apt-get update -y
    apt-get install -y docker.io docker-compose-plugin
    systemctl enable docker
    systemctl start docker
    usermod -aG docker ubuntu'
```

Wait for VM to be RUNNING:
```bash
gcloud compute instances describe kafka-vm \
  --project=project-2f9d2775-493e-4e59-9b8 \
  --zone=asia-south1-a \
  --format="value(status)"
```

**Step B — Get the VM internal IP:**
```bash
VM_INTERNAL_IP=$(gcloud compute instances describe kafka-vm \
  --project=project-2f9d2775-493e-4e59-9b8 \
  --zone=asia-south1-a \
  --format="value(networkInterfaces[0].networkIP)")
echo "VM internal IP: $VM_INTERNAL_IP"
```

**Step C — Add firewall rule to allow Kafka port 9092 within VPC (internal only):**
```bash
gcloud compute firewall-rules create allow-kafka-internal \
  --project=project-2f9d2775-493e-4e59-9b8 \
  --network=default \
  --allow=tcp:9092,tcp:29092 \
  --source-ranges=10.0.0.0/8 \
  --description="Kafka internal access from Cloud Run VPC connector"
```

**Step D — SCP vm-docker-compose.yml to the VM:**
```bash
gcloud compute scp payment-backend/infra/vm-docker-compose.yml \
  ubuntu@kafka-vm:/home/ubuntu/docker-compose.yml \
  --project=project-2f9d2775-493e-4e59-9b8 \
  --zone=asia-south1-a
```

**Step E — SSH into VM, configure Docker auth, set env vars, start services:**
```bash
gcloud compute ssh ubuntu@kafka-vm \
  --project=project-2f9d2775-493e-4e59-9b8 \
  --zone=asia-south1-a \
  --command='
    # Auth Docker to pull from Artifact Registry
    gcloud auth configure-docker asia-south1-docker.pkg.dev --quiet

    # Set required env vars (substitute real values from Plan 01)
    export VM_INTERNAL_IP=$(curl -s "http://metadata.google.internal/computeMetadata/v1/instance/network-interfaces/0/ip" -H "Metadata-Flavor: Google")
    export CLOUD_SQL_HOST=PLACEHOLDER_FROM_PLAN_01
    export REDIS_HOST=PLACEHOLDER_FROM_PLAN_01
    export GCS_BUCKET_NAME=PLACEHOLDER_FROM_PLAN_01

    # Write .env file for docker compose to pick up
    cat > /home/ubuntu/.env <<EOF
VM_INTERNAL_IP=$VM_INTERNAL_IP
CLOUD_SQL_HOST=$CLOUD_SQL_HOST
REDIS_HOST=$REDIS_HOST
GCS_BUCKET_NAME=$GCS_BUCKET_NAME
EOF

    # Start the stack
    cd /home/ubuntu
    docker compose up -d
  '
```

IMPORTANT: Replace PLACEHOLDER_FROM_PLAN_01 values with actual IPs/bucket names captured during Plan 01.
