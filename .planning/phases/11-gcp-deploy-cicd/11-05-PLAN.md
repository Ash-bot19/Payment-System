Branch note: D-26 specifies 'main' but this repo's canonical branch is 'master' (confirmed in
git status and recent commits). The workflow uses 'master'. This is correct for this repo.

Note: KAFKA_BOOTSTRAP_SERVERS is set via --set-env-vars on every deploy step using the
VM_INTERNAL_IP secret. This ensures the env var is correct on each pipeline run, not just on
the initial service creation (which used a placeholder value in Plan 01/03). The VM_INTERNAL_IP
secret must be set in GitHub (done in Task 1, Step E).

D-29 note: VM_INTERNAL_IP is a 6th secret added beyond D-29 to enable KAFKA_BOOTSTRAP_SERVERS
injection at pipeline deploy time. D-29 lists 5 secrets; this addition is intentional and
required for Cloud Run services to reach Kafka on the Compute Engine VM.

Note: Cloud Run does NOT inherit secrets from the previous revision when --set-secrets is
omitted. Every deploy step must include --set-secrets explicitly so each new revision has
access to Secret Manager secrets. Both --set-env-vars (for KAFKA_BOOTSTRAP_SERVERS) and
--set-secrets (for service credentials) are required on every deploy step.

Note: The repository root is one level above payment-backend/. All Docker build contexts use
`./payment-backend` as the build context.

```yaml
name: CI/CD Pipeline

on:
  push:
    branches: [master]
  pull_request:
    branches: [master]

env:
  PROJECT_ID: project-2f9d2775-493e-4e59-9b8
  REGION: asia-south1
  AR_REPO: ${{ secrets.AR_REPO }}

jobs:
  # ── Job 1: Build + Test ──────────────────────────────────────────────────────
  build-and-test:
    name: Build and Test
    runs-on: ubuntu-latest

    services:
      postgres:
        image: postgres:16-alpine
        env:
          POSTGRES_USER: payment
          POSTGRES_PASSWORD: payment
          POSTGRES_DB: payment_db
        options: >-
          --health-cmd pg_isready
          --health-interval 10s
          --health-timeout 5s
          --health-retries 5
        ports:
          - 5432:5432

    steps:
      - name: Checkout code
        uses: actions/checkout@v4

      - name: Set up Python 3.11
        uses: actions/setup-python@v5
        with:
          python-version: "3.11"
          cache: pip

      - name: Install dependencies
        working-directory: payment-backend
        run: |
          pip install --upgrade pip
          pip install -r requirements.txt

      - name: Run unit tests
        working-directory: payment-backend
        env:
          DATABASE_URL: ${{ secrets.DATABASE_URL }}
          DATABASE_URL_SYNC: postgresql://payment:payment@localhost:5432/payment_db
          REDIS_URL: redis://localhost:6379/0
          STRIPE_WEBHOOK_SECRET: ${{ secrets.STRIPE_WEBHOOK_SECRET }}
          KAFKA_BOOTSTRAP_SERVERS: localhost:9092
          ML_MODEL_PATH: ml/models/model.ubj
          GCP_PROJECT_ID: ""
          ENVIRONMENT: test
        run: |
          pytest tests/unit/ -x --timeout=60 -q

      - name: Set image tag
        id: tag
        run: echo "sha_short=$(git rev-parse --short HEAD)" >> $GITHUB_OUTPUT

    outputs:
      sha_short: ${{ steps.tag.outputs.sha_short }}

  # ── Job 2: Push Docker Images ─────────────────────────────────────────────────
  push-images:
    name: Push Docker Images
    runs-on: ubuntu-latest
    needs: build-and-test
    if: github.ref == 'refs/heads/master' && github.event_name == 'push'

    steps:
      - name: Checkout code
        uses: actions/checkout@v4

      - name: Authenticate to GCP
        uses: google-github-actions/auth@v2
        with:
          credentials_json: ${{ secrets.GCP_SA_KEY }}

      - name: Set up Cloud SDK
        uses: google-github-actions/setup-gcloud@v2

      - name: Configure Docker for Artifact Registry
        run: gcloud auth configure-docker asia-south1-docker.pkg.dev --quiet

      - name: Build and push webhook-service
        working-directory: payment-backend
        run: |
          docker build -f infra/Dockerfile.webhook \
            -t ${{ env.AR_REPO }}/webhook-service:${{ needs.build-and-test.outputs.sha_short }} \
            -t ${{ env.AR_REPO }}/webhook-service:latest \
            .
          docker push ${{ env.AR_REPO }}/webhook-service:${{ needs.build-and-test.outputs.sha_short }}
          docker push ${{ env.AR_REPO }}/webhook-service:latest

      - name: Build and push ml-scoring-service
        working-directory: payment-backend
        run: |
          docker build -f infra/Dockerfile.ml-service \
            -t ${{ env.AR_REPO }}/ml-scoring-service:${{ needs.build-and-test.outputs.sha_short }} \
            -t ${{ env.AR_REPO }}/ml-scoring-service:latest \
            .
          docker push ${{ env.AR_REPO }}/ml-scoring-service:${{ needs.build-and-test.outputs.sha_short }}
          docker push ${{ env.AR_REPO }}/ml-scoring-service:latest

      - name: Build and push streamlit-dashboard
        working-directory: payment-backend
        run: |
          docker build -f dashboard/Dockerfile \
            -t ${{ env.AR_REPO }}/streamlit-dashboard:${{ needs.build-and-test.outputs.sha_short }} \
            -t ${{ env.AR_REPO }}/streamlit-dashboard:latest \
            .
          docker push ${{ env.AR_REPO }}/streamlit-dashboard:${{ needs.build-and-test.outputs.sha_short }}
          docker push ${{ env.AR_REPO }}/streamlit-dashboard:latest

  # ── Job 3: Deploy to Staging (auto) ──────────────────────────────────────────
  deploy-staging:
    name: Deploy to Staging
    runs-on: ubuntu-latest
    needs: [build-and-test, push-images]
    if: github.ref == 'refs/heads/master' && github.event_name == 'push'
    environment: staging

    steps:
      - name: Authenticate to GCP
        uses: google-github-actions/auth@v2
        with:
          credentials_json: ${{ secrets.GCP_SA_KEY }}

      - name: Set up Cloud SDK
        uses: google-github-actions/setup-gcloud@v2

      - name: Deploy webhook-service to staging
        run: |
          gcloud run deploy webhook-service-staging \
            --image=${{ env.AR_REPO }}/webhook-service:${{ needs.build-and-test.outputs.sha_short }} \
            --region=${{ env.REGION }} \
            --project=${{ env.PROJECT_ID }} \
            --platform=managed \
            --set-env-vars="KAFKA_BOOTSTRAP_SERVERS=${{ secrets.VM_INTERNAL_IP }}:9092" \
            --set-secrets="STRIPE_WEBHOOK_SECRET=STRIPE_WEBHOOK_SECRET:latest,STRIPE_API_KEY=STRIPE_API_KEY:latest,DATABASE_URL=DATABASE_URL:latest,REDIS_URL=REDIS_URL:latest" \
            --quiet

      - name: Deploy ml-scoring-service to staging
        run: |
          gcloud run deploy ml-scoring-service-staging \
            --image=${{ env.AR_REPO }}/ml-scoring-service:${{ needs.build-and-test.outputs.sha_short }} \
            --region=${{ env.REGION }} \
            --project=${{ env.PROJECT_ID }} \
            --platform=managed \
            --set-env-vars="KAFKA_BOOTSTRAP_SERVERS=${{ secrets.VM_INTERNAL_IP }}:9092" \
            --set-secrets="REDIS_URL=REDIS_URL:latest" \
            --quiet

  # ── Job 4: Deploy to Production (manual gate) ─────────────────────────────────
  deploy-production:
    name: Deploy to Production
    runs-on: ubuntu-latest
    needs: [build-and-test, push-images, deploy-staging]
    if: github.ref == 'refs/heads/master' && github.event_name == 'push'
    environment: production

    steps:
      - name: Authenticate to GCP
        uses: google-github-actions/auth@v2
        with:
          credentials_json: ${{ secrets.GCP_SA_KEY }}

      - name: Set up Cloud SDK
        uses: google-github-actions/setup-gcloud@v2

      - name: Deploy webhook-service to production
        run: |
          gcloud run deploy webhook-service \
            --image=${{ env.AR_REPO }}/webhook-service:${{ needs.build-and-test.outputs.sha_short }} \
            --region=${{ env.REGION }} \
            --project=${{ env.PROJECT_ID }} \
            --platform=managed \
            --set-env-vars="KAFKA_BOOTSTRAP_SERVERS=${{ secrets.VM_INTERNAL_IP }}:9092" \
            --set-secrets="STRIPE_WEBHOOK_SECRET=STRIPE_WEBHOOK_SECRET:latest,STRIPE_API_KEY=STRIPE_API_KEY:latest,DATABASE_URL=DATABASE_URL:latest,REDIS_URL=REDIS_URL:latest" \
            --quiet

      - name: Deploy ml-scoring-service to production
        run: |
          gcloud run deploy ml-scoring-service \
            --image=${{ env.AR_REPO }}/ml-scoring-service:${{ needs.build-and-test.outputs.sha_short }} \
            --region=${{ env.REGION }} \
            --project=${{ env.PROJECT_ID }} \
            --platform=managed \
            --set-env-vars="KAFKA_BOOTSTRAP_SERVERS=${{ secrets.VM_INTERNAL_IP }}:9092" \
            --set-secrets="REDIS_URL=REDIS_URL:latest" \
            --quiet

      - name: Deploy streamlit-dashboard to production
        run: |
          gcloud run deploy streamlit-dashboard \
            --image=${{ env.AR_REPO }}/streamlit-dashboard:${{ needs.build-and-test.outputs.sha_short }} \
            --region=${{ env.REGION }} \
            --project=${{ env.PROJECT_ID }} \
            --platform=managed \
            --set-secrets="DATABASE_URL=DATABASE_URL:latest" \
            --quiet
```

After creating the file, also create the GitHub Actions environments with the production gate.

Run these gh CLI commands to set up environments:
```bash
REPO=$(gh repo view --json nameWithOwner -q .nameWithOwner)

# Create staging environment (no approval needed — auto-deploy)
gh api repos/$REPO/environments/staging \
  --method PUT \
  --field wait_timer=0

# Create production environment WITH required reviewer (manual approval)
# This requires the user's GitHub username or team
gh api repos/$REPO/environments/production \
  --method PUT \
  --field wait_timer=0 \
  --field "reviewers[][type]=User" \
  --field "reviewers[][id]=$(gh api user --jq .id)"
```

The `environment: production` in the workflow + reviewer requirement = the manual gate that
pauses the deploy-production job until the user clicks "Approve" in GitHub Actions UI.
